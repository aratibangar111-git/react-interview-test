
function CustomHook(){
	const[data,setdata]  = useState(null);
	
useEffect(()=>
	fetch("https://jsonplaceholder.typicode.com/posts")
.then(res=>res(json))
.then((data)=>setdata(data));
,[]);
	
	return(
	<>
		{data && data.map((d)=>{
			<p key = {d.id}>{d.id}</p>
			<p>{d.title}</p>
			<p>{d.body}</p>
		}
		)}
	</>
);
};
